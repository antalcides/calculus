\begin{align*} \lim\limits_{h\to0}\frac{5h^2+3}{2-3h^2}&=\frac{\lim\limits_{h\to0}\left(5h^2+3\right)}{\lim\limits_{h\to0}\left(2-3h^2\right)}&&\text{LL A5}\\ &=\frac{\lim\limits_{h\to0}\left(5h^2\right)+\lim\limits_{h\to0}3}{\lim\limits_{h\to0}2-\lim\limits_{h\to0}\left(3h^2\right)}&&\text{LL A1 and A2}\\ &=\frac{5\lim\limits_{h\to0}h^2+\lim\limits_{h\to0}3}{\lim\limits_{h\to0}2-3\lim\limits_{h\to0}h^2}&&\text{LL A3}\\ &=\frac{5\left(\lim\limits_{h\to0}h\right)^2+\lim\limits_{h\to0}3}{\lim\limits_{h\to0}2-3\left(\lim\limits_{h\to0}h\right)^2}&&\text{LL A6}\\ &=\frac{5\cdot0^2+3}{2-3\cdot0^2}&&\text{LL R1 and R2}\\ &=\frac{3}{2}&&\text{This limit does exist.} \end{align*}
